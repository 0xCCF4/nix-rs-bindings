1use std::env;
2use std::path::PathBuf;
3use pkg_config::Library;
4
5pub fn configure_library<F: FnOnce(&Library, bindgen::Builder) -> bindgen::Builder>(
6    pkg_name: &str,
7    static_config_environment_overwrite: &str,
8    feature_static: bool,
9    feature_shared: bool,
10    configure_bindings: F,
11) -> PathBuf {
12    let env_override = env::var(static_config_environment_overwrite)
13        .map(|value| value == "1" || value == "static" || value == "true")
14        .ok();
15
16    #[allow(unreachable_code)]
17    let static_link = if let Some(static_overwrite) = env_override {
18        static_overwrite
19    } else {
20        if feature_shared && feature_static {
21            panic!(
22                "Both features \"shared\" and \"static\" are set at the same time. This is forbidden. You may override the settings via specifying the NIX_API_STATIC environment variable, which is prioritized over build features."
23            );
24        } else if !feature_shared && !feature_static {
25            println!(
26                "cargo:warning=Neither feature 'shared' nor feature 'static' specified -> defaulting to 'shared'."
27            );
28        }
29        feature_static
30    };
31
32    let library = pkg_config::Config::new()
33        .statik(static_link)
34        .probe(pkg_name)
35        .unwrap();
36
37    for link_path in &library.link_paths {
38        println!(
39            "cargo:rustc-link-search=native={}",
40            link_path.to_str().expect("non-UTF-8 path")
41        );
42    }
43
44    for name in &library.libs {
45        println!(
46            "cargo:rustc-link-lib={}{}",
47            if static_link { "static=" } else { "" },
48            name
49        );
50    }
51
52    let bindings = bindgen::Builder::default()
53        // defines
54        .clang_args(library.defines.iter().map(|(name, value)| {
55            if let Some(value) = value {
56                format!("-D{name}={value}")
57            } else {
58                format!("-D{name}")
59            }
60        })).header_contents("stub.h", "");
61
62    let bindings = configure_bindings(&library, bindings);
63
64    let bindings = bindings
65        // Include paths
66        .clang_arg(format!("-I{}", env::var("OUT_DIR").unwrap()))
67        .clang_args(
68            library
69                .include_paths
70                .iter()
71                .map(|path| format!("-I{}", path.to_str().expect("path is valid UTF-8"))),
72        )
73        // Tell cargo to invalidate the built crate whenever any of the
74        // included header files changed.
75        .parse_callbacks(Box::new(bindgen::CargoCallbacks::new()))
76        // Finish the builder and generate the bindings.
77        .generate()
78        // Unwrap the Result and panic on failure.
79        .expect("Unable to generate bindings");
80
81    let out_path = PathBuf::from(env::var("OUT_DIR").unwrap()).join("bindings.rs");
82    bindings
83        .write_to_file(&out_path)
84        .expect("Couldn't write bindings!");
85
86    out_path
87}
